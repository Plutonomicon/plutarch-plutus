Pattern matching over Data inside PLC.

Either

\(d : data) ->
    /\(r :: *) ->
     \(fConstr : integer -> list data -> r)
      (fMap : list (pair data data) -> r)
      (fList : list data -> r)
      (fI : integer -> r)
      (fB : bytestring -> r) ->
          matchData
              {r}
              fConstr
              fMap
              fList
